> git add [file name]
> git add .
> git commit
> git status
> git log
# 一行で表示する
    > git log --oneline
#ファイルの差分を表示する
    > git log -p index.html
#行数を指定してログを表示
> git log -n 3
> git diff
#ステージとコミットの差分
    > git diff HEAD
> git remote add origin https://github.com/yamn15/intro_git
> git push origin master

> gitignoreファイルの書き方
# #から始まる行はコメント
# 指定したファイルを除外
    index.html
#ルートディレクトリを指定
    /root.html
#ディレクトリ以下を除外
    /dir

#コミットしたファイルを管理から外す
#ファイルも一緒に削除
    > git rm [file name]
#ディレクトリも一緒に削除
    > git rm -r [file name]
#ファイルは残す
    > git rm --cached [file name]
